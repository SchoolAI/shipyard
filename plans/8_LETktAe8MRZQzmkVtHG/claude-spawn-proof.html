Spawn Implementation (apps/daemon/src/agent-spawner.ts)

export function spawnClaudeCode(opts: SpawnAgentOptions): ChildProcess {
  const { taskId, prompt, cwd } = opts;

  // Stop existing agent if running
  if (activeAgents.has(taskId)) {
    stopAgent(taskId);
  }

  const child = spawn(
    'claude',
    ['-p', prompt, '--allowedTools', 'mcp__shipyard__*'],
    {
      cwd,
      stdio: ['ignore', 'pipe', 'pipe'],
      env: {
        ...process.env,
        SHIPYARD_TASK_ID: taskId,
      },
    }
  );

  // Track the agent
  if (child.pid) {
    activeAgents.set(taskId, { taskId, process: child, pid: child.pid });
  }

  // Stream stdout/stderr to WebSocket clients
  child.stdout?.on('data', (data) => { /* streamed via protocol */ });
  child.stderr?.on('data', (data) => { /* streamed via protocol */ });

  return child;
}
